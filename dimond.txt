#include <stdio.h>
int main() {
    int i, j, k, n;
    n = 4;

    printf("Diamond shape with numbers:\n");

    // Upper half of diamond
    for(i = 1; i <= n; i++) {
        for(j = i; j < n; j++)
            printf(" ");
        for(k = 1; k <= i; k++)
            printf("%d ", i);
        printf("\n");
    }

    // Lower half of diamond
    for(i = n - 1; i >= 1; i--) {
        for(j = n; j > i; j--)
            printf(" ");
        for(k = 1; k <= i; k++)
            printf("%d ", i);
        printf("\n");
    }

    return 0;
}
